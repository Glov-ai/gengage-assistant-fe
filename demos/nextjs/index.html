Next.js + Gengage Integration

This page documents how to integrate Gengage widgets into a Next.js app. Next.js requires a project build — there is no single-HTML demo.

Gengage widgets are vanilla DOM — they work in any framework. The key requirement in Next.js is to initialise them client-side only (not during SSR) by using "use client" and useEffect.

1 Install

npm install @gengage/assistant-fe

2 Create a client component

File: components/GengageWidgets.tsx 

"use client";

import { useEffect, useRef } from "react";
import { initOverlayWidgets } from "@gengage/assistant-fe";
import type { OverlayWidgetsController } from "@gengage/assistant-fe";

interface Props {
  accountId: string;
  middlewareUrl: string;
  sku?: string;
  locale?: string;
}

export function GengageWidgets({ accountId, middlewareUrl, sku, locale = "en" }: Props) {
  const controllerRef = useRef<OverlayWidgetsController | null>(null);

  useEffect(() => {
    let cancelled = false;

    initOverlayWidgets({
      accountId,
      middlewareUrl,
      locale,
      sku,
      pageContext: { pageType: sku ? "pdp" : "other", sku },
      chat: { variant: "floating" },
      qna: { mountTarget: "#gengage-qna" },
      simrel: { mountTarget: "#gengage-simrel" },
      onAddToCart: (p) => console.log("add-to-cart", p),
      onProductNavigate: (url) => { window.location.href = url; },
    }).then((ctrl) => {
      if (!cancelled) controllerRef.current = ctrl;
    });

    return () => {
      cancelled = true;
      controllerRef.current?.destroy();
    };
  }, [accountId, middlewareUrl, sku, locale]);

  return (
    <>
      <div id="gengage-qna" />
      <div id="gengage-simrel" />
    </>
  );
}

3 Use in a page

File: app/products/[sku]/page.tsx 

import { GengageWidgets } from "@/components/GengageWidgets";

export default function ProductPage({ params }: { params: { sku: string } }) {
  return (
    <main>
      <h1>Product {params.sku}</h1>
      <GengageWidgets
        accountId="your-account-id"
        middlewareUrl="https://YOUR-BACKEND.gengage.ai"
        sku={params.sku}
      />
    </main>
  );
}

4 SPA navigation

When the user navigates between products, React will re-render the component with a new sku prop. The useEffect cleanup destroys the old widgets and re-initialises with the new SKU.

For finer control, call controller.updateContext() instead of full re-init.
